-- Function to handle monthly rollover
CREATE OR REPLACE FUNCTION handle_monthly_rollover(p_user_id TEXT)
RETURNS void AS $$
DECLARE
  v_tier TEXT;
  v_current_balance INTEGER;
  v_monthly_credits INTEGER;
  v_rollover_limit INTEGER;
  v_new_balance INTEGER;
BEGIN
  -- Get user info
  SELECT 
    subscription_tier, 
    credit_balance 
  INTO v_tier, v_current_balance
  FROM user_profiles
  WHERE id::text = p_user_id;
  
  -- Get tier limits
  SELECT 
    monthly_credits,
    (features->>'rollover')::INTEGER
  INTO v_monthly_credits, v_rollover_limit
  FROM tier_limits
  WHERE tier = v_tier;
  
  -- Calculate rollover (unused credits)
  v_new_balance := LEAST(
    v_current_balance + v_monthly_credits,  -- Total
    v_monthly_credits + v_rollover_limit    -- Max allowed
  );
  
  -- Update balance
  UPDATE user_profiles
  SET 
    credit_balance = v_new_balance,
    accumulated_credits = v_current_balance  -- Track rollover amount
  WHERE id::text = p_user_id;
  
  -- Log
  INSERT INTO credit_usage (user_id, action, credits_used, credits_remaining, metadata)
  VALUES (
    p_user_id, 
    'monthly_rollover', 
    -v_monthly_credits,  -- Negative = credit added
    v_new_balance,
    jsonb_build_object(
      'previous_balance', v_current_balance,
      'monthly_grant', v_monthly_credits,
      'rollover_amount', v_current_balance
    )
  );
END;
$$ LANGUAGE plpgsql;

// Queue system
interface QueueJob {
  id: string;
  userId: string;
  tier: string;
  priorityScore: number;
  createdAt: Date;
  estimatedWait: number; // seconds
}

class PriorityQueue {
  private queue: QueueJob[] = [];
  
  add(job: QueueJob) {
    this.queue.push(job);
    // Sort by priority (highest first), then by time (oldest first)
    this.queue.sort((a, b) => {
      if (a.priorityScore !== b.priorityScore) {
        return b.priorityScore - a.priorityScore; // Descending
      }
      return a.createdAt.getTime() - b.createdAt.getTime(); // Ascending
    });
  }
  
  getPosition(jobId: string): number {
    return this.queue.findIndex(j => j.id === jobId) + 1;
  }
  
  getEstimatedWait(jobId: string): number {
    const position = this.getPosition(jobId);
    const avgProcessTime = 180; // 3 minutes per job
    return position * avgProcessTime;
  }
}
